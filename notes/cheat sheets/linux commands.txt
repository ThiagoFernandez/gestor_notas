LINUX COMMAND CHEAT SHEET
=========================

--------------------------------------------------
COMANDOS BASICOS
--------------------------------------------------

pwd
mostrar directorio actual

ls
listar archivos

ls -l
listar en formato detallado

ls -a
mostrar archivos ocultos

ls -lh
mostrar tamaño legible

clear
limpiar terminal

history
mostrar historial de comandos


--------------------------------------------------
NAVEGACION
--------------------------------------------------

cd carpeta
entrar a carpeta

cd ..
subir un nivel

cd ~
ir al home

cd /
ir a la raíz

cd -
volver al directorio anterior


--------------------------------------------------
MANEJO DE ARCHIVOS
--------------------------------------------------

touch archivo.txt
crear archivo

cp origen destino
copiar archivo

cp archivo carpeta/
copiar a carpeta

mv origen destino
mover o renombrar

rm archivo
borrar archivo

rm -r carpeta
borrar carpeta

rm -rf carpeta
borrar carpeta forzado


--------------------------------------------------
CREAR DIRECTORIOS
--------------------------------------------------

mkdir carpeta

mkdir -p carpeta/subcarpeta


--------------------------------------------------
VER CONTENIDO DE ARCHIVOS
--------------------------------------------------

cat archivo.txt
mostrar contenido

less archivo.txt
leer archivo navegable

head archivo.txt
primeras 10 líneas

head -n 20 archivo.txt

tail archivo.txt
últimas líneas

tail -f log.txt
ver archivo en tiempo real

--------------------------------------------------
ARCH LINUX / ENDEAVOUROS
--------------------------------------------------

PACMAN (GESTOR DE PAQUETES)

actualizar sistema

sudo pacman -Syu


instalar paquete

sudo pacman -S paquete


eliminar paquete

sudo pacman -R paquete


eliminar paquete y dependencias

sudo pacman -Rns paquete


buscar paquete

pacman -Ss paquete


ver paquetes instalados

pacman -Q


ver info de paquete

pacman -Qi paquete


limpiar cache

sudo pacman -Sc


--------------------------------------------------
AUR (yay)
--------------------------------------------------

instalar paquete AUR

yay -S paquete


actualizar todo

yay -Syu


buscar paquete

yay -Ss paquete


eliminar paquete

yay -Rns paquete

--------------------------------------------------

sudo systemctl stop servicio

sudo systemctl enable servicio


deshabilitar servicio

sudo systemctl disable servicio


ver estado de servicio

systemctl status servicio
